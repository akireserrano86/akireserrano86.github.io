Temperature Conversion

This ruby program will prompt the user for a temperature in degrees Celsius and let the user know what the corresponding temperature is in Fahrenheit. 


          def convert(celsius)
            return celsius * 1.8 + 32
          end

          puts "Enter Temperature in Celsius:"
          celsius = gets.chomp.to_f
          farenheit = convert(celsius)
          puts "#{celsius} Celsius is #{farenheit} Farenheit"
        


            def calc_line(num)

              if num % 10 == 1
                  puts "That's the #{num}st item!"
              elsif num % 10 == 2
                  puts "That's the #{num}nd item!"
              elsif num % 10 == 3
                  puts "That's the #{num}rd item!"
              else

              end

            end

            def calc_ordinal(num)

              if num % 100 < 11
                calc_line(num)
              else
                if num <= 99
                  if num == 11 || num == 12 || num == 13
                    puts "That's the #{num}th item!"
                  else
                    calc_line(num)
                  end
                else
                  puts "That's the #{num}th item!"
                end
              end

            end

            puts "Enter number:"
            num = gets.chomp.to_i
            calc_ordinal(num)

Foobar

This ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 


          def foobar(seq)
            foo = []
            num = 1

            while num <= seq
              if num % 3 == 0 && num % 5 == 0
                  foo << "foobar"
              elsif num % 5 == 0
                  foo << "bar"
              elsif num % 3 == 0
                  foo << "foo"
              else
                  foo << num
              end
              num = num + 1
            end
            puts foo.inspect
          end

            puts "How many digits in the sequence?"
            seq = gets.chomp.to_i

            foobar(seq)
